
namespace = DPFF
# Timer for Pre-FTL buildings to repair.
planet_event = {
	id = DPFF.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_owner = yes
		owner = { is_primitive = yes }
	}

	immediate = {
		planet_event = { id = DPFF.3 months = 12 random = 6 }
	}
}

namespace = DEI
# Tried to keep Merc Army replenishment rates close to vanilla.
country_event = {
	id = DEI.10
	hide_window = yes
	# is_triggered_only = yes
	trigger = {
		is_country_type = enclave_mercenary
		has_civic = civic_mercenary_enclave
	}
	# TODO DEPRECATE: optimized alternative
	mean_time_to_happen = {
		days = 72 # Average time for a Merc Army.
		modifier = {
			factor = 0.5
			mid_game_years_passed > 0 # They make more armies after midgame.
		}
	}

	immediate = {
		if = {
			limit = {
				check_variable = { which = mercenary_armies_for_hire value < 30 }
			}
			change_variable = { which = mercenary_armies_for_hire value = 1 }
		}
		if = {
			limit = {
				has_modifier = mercenary_fleet_size_upgrade_4
				check_variable = { which = mercenary_armies_for_hire value < 15 }
			}
			change_variable = { which = mercenary_mecha_armies_for_hire value = 1 }
		}
	}
}
