# Timer for Pre-FTL buildings to repair.
planet_event = {
	id = DPFF.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_owner = yes
		owner = { is_primitive = yes }
	}

	immediate = {
		planet_event = { id = DPFF.3 months = 12 random = 6 }
	}
}
