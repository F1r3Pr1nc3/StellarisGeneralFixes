## Vfix empty rebel war goal (for unknown reason) by FirePrince

event = {
	id = rebellionfix.33
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		# years_passed > 20
		any_playable_country = {
			is_subject = yes
			is_at_war = yes
			any_war = {
				is_war_participant = { who = prev side = attackers }
				any_defender = { is_same_value = prevprev.overlord }
				using_war_goal = { owner = prev }
			}
		}
	}

	immediate = {
		# TODO: can be extended for other war_goals
		random_playable_country = {
			limit = {
				is_subject = yes
				is_at_war = yes
				any_war = {
					is_war_participant = { who = prev side = attackers }
					any_defender = { is_same_value = prevprev.overlord }
					using_war_goal = { owner = prev }
				}
				is_war_leader = yes
			}
			weights = {
				base = 1
				modifier = { add = 99 is_ai = no }
			}
			random_war = {
				limit = {
					is_war_participant = { who = prev side = attackers }
					any_defender = { is_same_value = prevprev.overlord }
					using_war_goal = { owner = prev }
				}
				set_war_goal = {
					type = wg_independence
					target = prev
					enemy = prev.overlord
				}
			}
		}
	}
}

			limit = {
				any_owned_species = { is_same_species = root has_trait = trait_syncretic_proles }
			}
			## Get majority rebel species
			every_owned_species = {
				limit = {
					is_same_species = root
					has_trait = trait_syncretic_proles
				}
				export_trigger_value_to_variable = {
					trigger = count_species_pop_group
					variable = var_species_pops_percent
					parameters = {
						limit = {
							is_exact_same_species = prevprev
							pop_group_has_trait = trait_syncretic_proles
							is_owned_by = root
						}
					}
				}
			}
			ordered_owned_species = {
				limit = { is_variable_set = var_species_pops_percent }
				position = 0
				order_by = var_species_pops_percent
				save_event_target_as = syncretic_species
			}
			export_trigger_value_to_variable = {
				trigger = count_owned_pop_amount
				variable = syncretic_pops
				parameters = {
					limit = { is_exact_same_species = event_target:syncretic_species }
				}
			}
			owner_species = {
				if = {
					limit = {
						NOR = {
							has_trait = trait_syncretic_proles
							is_variable_set = var_species_pops_percent
						}
					}
					export_trigger_value_to_variable = {
						trigger = count_species_pop_group

							limit = {

							}
						}
					}
					if = {
						limit = {
							check_variable = {
								which = var_species_pops_percent
								value > event_target:syncretic_species.var_species_pops_percent
							}
						}
						clear_variable = var_species_pops_percent
					}
				}
			}
			if = {
				limit = {
					owner_species = { is_variable_set = var_species_pops_percent }
				}
				modify_species = {
					species = event_target:syncretic_species
					remove_trait = trait_syncretic_proles

							change_dominant_species = { species = prev change_all = yes }
						}

				}
			}

				count = syncretic_pops # value:fifty_percent_species
				weighted_random_owned_pop_group = {
					limit = {
						is_same_species = owner_species
						NOR = {
							is_exact_same_species = event_target:syncretic_species
							has_trait = trait_syncretic_proles
						}
						OR = {
							is_enslaved = yes
							is_unemployed = yes
							has_citizenship_rights = no
							is_pop_category = worker
						}
					}
					change_species = event_target:syncretic_species
				}
			}
			# log = "FIX: Changed rebelling syncretic species \\[Root.owner_species.GetName] from \\[Root.GetName]!"
			clear_variable = syncretic_pops
			every_owned_species = {
				limit = {
					is_same_species = root
					is_variable_set = var_species_pops_percent
				}
				clear_variable = var_species_pops_percent
			}
		}
		if = {
			limit = {
				is_gestalt = yes
				num_ethics > 1
			}
			clear_ethos = yes
			shift_ethic = ethic_gestalt_consciousness
			break = yes
		}

				}
