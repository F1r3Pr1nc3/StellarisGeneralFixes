### Crisis Events II (extradimensionals)
# Portal Destroyed (HIDDEN on_fleet_destroyed_victim)
# Vfix:  (action gate for crisis.1012, crisis.1115, crisis.1215)
country_event = {
	id = crisis.1015
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_global_flag = extradimensional_invasion_happened # Vfix opt
		OR = {
			is_country_type = extradimensional
			is_country_type = extradimensional_2
			is_country_type = extradimensional_3
			is_country_type = portal_holder # Vfix for safety
		}
		fromfrom = { is_ship_size = dimensional_portal_ed }
	}

	immediate = {
		log = "SGF: [This.GetName] Portal Destroyed"
		switch = {
			trigger = is_country_type
			extradimensional	= { country_event = { id = crisis.1012 } }
			extradimensional_2	= { country_event = { id = crisis.1115 } }
			extradimensional_3	= { country_event = { id = crisis.1215 } }
			portal_holder = {
				fromfrom = {
					log = "SGF: unorthodox Portal Destroyed triggered!?"
					if = { limit = { any_owned_ship = { has_ship_flag = unbidden_portal } } event_target:extradimensionals = { country_event = { id = crisis.1012 } } }
					else_if = { limit = { any_owned_ship = { has_ship_flag = aberrant_portal } } event_target:extradimensionals2 = { country_event = { id = crisis.1115 } } }
					else_if = { limit = { any_owned_ship = { has_ship_flag = vehement_portal } } event_target:extradimensionals3 = { country_event = { id = crisis.1215 } } }
				}
			}
		}
		# crisis.1254 gets yearly triggered
	}
}

# Anchor Destroyed (HIDDEN on_starbase_destroyed)
starbase_event = {
	id = crisis.1284
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			has_starbase_size = starbase_exd_0
			has_starbase_size = starbase_exd
		}
		has_global_flag = extradimensional_invasion_happened
		exists = owner
		owner = {
			OR = {
				is_country_type = extradimensional
				is_country_type = extradimensional_2
				is_country_type = extradimensional_3
			}
		}
	}

	immediate = {
		if = {
			limit = {
				NOT = { exists = event_target:portal_holder_1 }
			}
			random_country = {
				limit = { is_country_type = portal_holder }
				save_global_event_target_as = portal_holder_1
			}
		}
		owner = {
			save_event_target_as = starbase_exd_owner
			switch = { # Vfix opt
				trigger = is_country_type
				extradimensional = {
					every_playable_country = {
						limit = { has_event_chain = "extradimensional_invasion_chain" }
						add_event_chain_counter = {
							event_chain = "extradimensional_invasion_chain"
							counter = "dimensional_anchors_1"
							amount = -1
						}
					}
					if = {
						limit = {
							NOT = { any_controlled_fleet = { is_ship_size = starbase_exd } }
						}
						event_target:portal_holder_1 = { # Vfix opt
							random_controlled_ship = {
								limit = {
									is_ship_size = dimensional_portal_ed
									has_ship_flag = unbidden_portal
								}
								fleet = { set_controller = root set_event_locked = no }
							}
						}
					}
				}
				extradimensional_2 = {
					every_playable_country = {
						limit = { has_event_chain = "extradimensional_invasion_chain_2" }
						add_event_chain_counter = {
							event_chain = "extradimensional_invasion_chain_2"
							counter = "dimensional_anchors_2"
							amount = -1
						}
					}
					if = {
						limit = {
							NOT = { any_controlled_fleet = { is_ship_size = starbase_exd } }
						}
						event_target:portal_holder_1 = { # Vfix opt
							random_controlled_ship = {
								limit = {
									is_ship_size = dimensional_portal_ed
									has_ship_flag = aberrant_portal
								}
								fleet = { set_controller = root set_event_locked = no }
							}
						}
					}
				}
				extradimensional_3 = {
					every_playable_country = {
						limit = { has_event_chain = "extradimensional_invasion_chain_3" }
						add_event_chain_counter = {
							event_chain = "extradimensional_invasion_chain_3"
							counter = "dimensional_anchors_3"
							amount = -1
						}
					}
					if = {
						limit = {
							NOT = { any_controlled_fleet = { is_ship_size = starbase_exd } }
						}
						event_target:portal_holder_1 = { # Vfix opt
							random_controlled_ship = {
								limit = {
									is_ship_size = dimensional_portal_ed
									has_ship_flag = vehement_portal
								}
								fleet = { set_controller = root set_event_locked = no }
							}
						}
					}
				}
			}
			if = {
				limit = {
					event_target:portal_holder_1 = {
						any_owned_ship = {
							is_ship_size = dimensional_portal_ed
							OR = {
								is_controlled_by = prevprev # vincible
								AND = { # Not protected: TODO: just one fleet?
									NOT = { is_within_borders_of = prevprev }
									OR = {
										AND = {
											prevprev = { has_country_flag = unbidden }
											has_ship_flag = unbidden_portal
										}
										AND = {
											prevprev = { has_country_flag = aberrant }
											has_ship_flag = aberrant_portal
										}
										AND = {
											prevprev = { has_country_flag = vehement }
											has_ship_flag = vehement_portal
										}
									}
								}
							}
						}
					}
				}
				country_event = { id = crisis.1611 } # Full Fallback to Portal
			}
		}
		# Remove System Effect
		solar_system = {
			log = "\\[event_target:starbase_exd_owner.GetName] anchor destroyed on \\[This.GetName] event crisis.1281 triggered"
			random_system_ambient_object = {
				limit = { has_ambient_object_flag = extradimensional_system_effect }
				destroy_ambient_object = this
			}
		}
	}

	# Vfix Double check (as it is too risky for very important event)
	after = {
		event_target:starbase_exd_owner = {
			if = {
				limit = {
					NOT = {
						any_controlled_fleet = {
							OR = {
								is_ship_size = starbase_exd
								is_ship_size = starbase_exd_0
								is_ship_size = dimensional_portal_ed
							}
						}
					}
				}
				# Vfix opt
				event_target:portal_holder_1 = {
					random_owned_ship = {
						limit = {
							is_ship_size = dimensional_portal_ed
							OR = {
								AND = {
									prevprev = { has_country_flag = unbidden }
									has_ship_flag = unbidden_portal
								}
								AND = {
									prevprev = { has_country_flag = aberrant }
									has_ship_flag = aberrant_portal
								}
								AND = {
									prevprev = { has_country_flag = vehement }
									has_ship_flag = vehement_portal
								}
							}
						}
						fleet = { set_controller = prevprev set_event_locked = no }
					}
				}
			}
		}
	}
}

# Full Fallback to Portal
country_event = {
	id = crisis.1611
	hide_window = yes
	is_triggered_only = yes

	trigger = {
	}

	immediate = {
		if = { # Vfix opt
			limit = {
				NOT = { exists = event_target:dimensional_portal_system }
			}
			random_controlled_fleet = {
				limit = { is_ship_size = dimensional_portal_ed }
				solar_system = { save_event_target_as = dimensional_portal_system }
			}
		}
		every_owned_fleet = {
			limit = {
				is_mobile = yes
				exists = solar_system
				NOR = {
					is_civilian = yes
					is_garrison = yes
					solar_system = {
						OR = {
							has_star_flag = extradimensional_origin_system
							has_star_flag = extradimensional_second_portal_system
							has_star_flag = extradimensional_third_portal_system
						}
					}
				}
			}
			solar_system = {
				prev = {
					if = {
						limit = { is_in_combat = yes }
						auto_move_to_planet = {
							target = event_target:dimensional_portal_system.star
							clear_auto_move_on_arrival = yes
						}
						log = "SGF: move back to portal system [This.GetName]"
					}
					else_if = {
						limit = { can_go_mia = yes }
						remove_auto_move_target = yes
						clear_fleet_actions = this
						set_mia = mia_return_home
						log = "SGF: go mia to portal system [This.GetName]"
					}
					else = {
						order_forced_return = yes
						log = "SGF: order forced return to portal system [This.GetName]"
					}
				}
			}
			# Assign leader to fleet (if missing)
			if = {
				limit = {
					NOT = { exists = leader }
					root = {
						any_owned_leader = {
							leader_class = commander
							is_idle = yes
							NOR = {
								exists = fleet
								is_councilor = yes
								is_ruler = yes
							}
						}
					}
				}
				root = {
					random_owned_leader = {
						limit = {
							leader_class = commander
							is_idle = yes
							NOR = {
								is_councilor = yes
								is_ruler = yes
								exists = fleet
							}
						}
						prevprev = { assign_leader = prev }
					}
				}
			}
		}
	}
}
