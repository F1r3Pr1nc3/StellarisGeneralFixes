namespace = machine_age
# on_single_player_save_game_load
# Vfix leader trait just for spiritualists
event = {
	id = machine_age.2151
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes

	immediate = {
		every_playable_country = {
			limit = { has_origin = origin_cybernetic_creed }
			country_event = { id = machine_age.2152 days = 3 }
		}
	}
}

# Vfix leader trait just for spiritualists
country_event = {
	id = machine_age.2152
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_origin = origin_cybernetic_creed
	}

	immediate = {
		# Remove
		every_owned_leader = {
			limit = {
				NOR = {
					is_spiritualist = yes
					is_ruler = yes
				}
				has_trait = leader_trait_ritualistic_implants_cyborg
			}
			remove_trait = leader_trait_ritualistic_implants_cyborg
			if = {
				limit = {
					species = { has_trait = trait_cyborg_ritualistic_implants }
				}
				prev = {
					random_owned_species = {
						limit = {
							is_same_species = prevprev.species
							NOT = { has_trait = trait_cyborg_ritualistic_implants }
						}
						weights = {
							base = 1
							modifier = { add = 1000 has_cybernetic_creed_trait = yes }
						}
						prevprev = { change_species = prev update_leader_after_modification = yes }
					}
				}
			}
		}
		every_pool_leader = {
			limit = {
				is_spiritualist = no
				has_trait = leader_trait_ritualistic_implants_cyborg
			}
			remove_trait = leader_trait_ritualistic_implants_cyborg
			if = {
				limit = {
					species = { has_trait = trait_cyborg_ritualistic_implants }
				}
				prev = {
					random_owned_species = {
						limit = {
							is_same_species = prevprev.species
							NOT = { has_trait = trait_cyborg_ritualistic_implants }
						}
						weights = {
							base = 1
							modifier = { add = 1000 has_cybernetic_creed_trait = yes }
						}
						prevprev = { change_species = prev update_leader_after_modification = yes }
					}
				}
			}
		}
		# Add
		every_owned_leader = {
			limit = {
				OR = {
					is_spiritualist = yes
					is_ruler = yes
				}
				NOR = {
					has_trait = leader_trait_ritualistic_implants_cyborg
					has_trait = leader_trait_psionic
					has_trait = leader_trait_chosen
					has_trait = leader_trait_cyborg
				}
			}
			add_trait = {
				trait = leader_trait_ritualistic_implants_cyborg
				show_message = no
			}
			if = {
				limit = {
					species = { has_cybernetic_creed_trait = no }
				}
				prev = {
					random_owned_species = {
						limit = {
							is_same_species = prevprev.species
							has_cybernetic_creed_trait = yes
						}
						weights = {
							base = 1
							modifier = {
								add = 1000
								has_trait = leader_trait_ritualistic_implants_cyborg
							}
						}
						prevprev = { change_species = prev update_leader_after_modification = yes }
					}
				}
			}
		}
		every_pool_leader = {
			limit = {
				is_spiritualist = yes
				NOR = {
					has_trait = leader_trait_ritualistic_implants_cyborg
					has_trait = leader_trait_psionic
					has_trait = leader_trait_chosen
					has_trait = leader_trait_cyborg
				}
			}
			add_trait = {
				trait = leader_trait_ritualistic_implants_cyborg
				show_message = no
			}
			if = {
				limit = {
					species = { has_cybernetic_creed_trait = no }
				}
				prev = {
					random_owned_species = {
						limit = {
							is_same_species = prevprev.species
							has_cybernetic_creed_trait = yes
						}
						weights = {
							base = 1
							modifier = {
								add = 1000
								has_trait = leader_trait_ritualistic_implants_cyborg
							}
						}
						prevprev = { change_species = prev update_leader_after_modification = yes }
					}
				}
			}
		}
	}
}
