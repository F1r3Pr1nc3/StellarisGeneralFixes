indiscriminate = {
	trigger = {
		# exists = owner
		owner = {
			is_crisis_faction = no # Vfix
			OR = {
				NOT = { is_country_type = default }
				has_policy_flag = orbital_bombardment_indiscriminate
				has_policy_flag = orbital_bombardment_armageddon
			}
		}
		# count_owned_ship = { count = 0 # Vfix opt
		# 	limit = {
		# 		has_ship_flag = behemoth_hatchling_ship
		# 	}
		# }
	}

	default = no

	stop_when_armies_dead = no
	stop_when_ground_combat = no
	accept_surrender = yes
	abduct_pops = no

	planet_damage = 1.0
	army_damage = 1.5

	kill_pop_chance = {
		base = 0.5
	}
	min_pops_to_kill_pop = 1100
	kill_pop_amount = { min = 0 max = 200 }
	# root = fleet
	# from = planet
	ai_weight = {
		weight = 5
		modifier = {
			factor = 0.01
			exists = from
			from = {
				NOT = { owner = { is_hostile = root.owner } }
				NAND = {
					controller = { is_crisis_faction = yes }
					planet_garrison_strength > 0
				}
			}
		}
		modifier = { # Vfix Ariphaos
			factor = 0
			owner = {
				has_policy_flag = orbital_bombardment_armageddon
				has_valid_civic = civic_machine_terminator
			}
		}
	}
}

armageddon = {
	trigger = {
		# exists = owner
		OR = {
			owner = {
				OR = {
					has_policy_flag = orbital_bombardment_armageddon
					is_crisis_faction = yes
					has_crisis_perk = menp_death_from_above # Menace Perk: Death From Above
				}
			}
			AND = {
				owner = { is_mercenary = yes }
				controller = {
					OR = {
						has_policy_flag = orbital_bombardment_armageddon
						is_crisis_faction = yes
						has_menace_perk = menp_death_from_above # Menace Perk: Death From Above
					}
				}
			}
		}
	}

	default = no

	stop_when_armies_dead = no
	stop_when_ground_combat = no
	abduct_pops = no

	planet_damage = 1.5
	army_damage = 2.0

	kill_pop_chance = {
		base = 1.0
	}
	min_pops_to_kill_pop = 0
	kill_pop_amount = { min = 0 max = 200 }

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 10
		modifier = {
			factor = 0.01
			exists = from
			from = {
				NOT = { owner = { is_hostile = root.owner } }
				NAND = {
					controller = { is_crisis_faction = yes }
					planet_garrison_strength > 0
				}
			}
			exists = owner
			owner = { is_crisis_faction = no }
		}
	}
}

raiding = {
	trigger = {
		# exists = owner
		owner = { # Vfix opt
			is_crisis_faction = no # Vfix
			OR = {
				has_ascension_perk = ap_nihilistic_acquisition
				has_valid_civic = civic_barbaric_despoilers
				has_origin = origin_slavers
				has_origin = origin_khan_successor
				is_country_type = debt_collectors
				AND = {
					is_mercenary = yes
					prev.controller = {
						OR = {
							has_ascension_perk = ap_nihilistic_acquisition
							has_valid_civic = civic_barbaric_despoilers
							has_origin = origin_slavers
							has_origin = origin_khan_successor
						}
					}
				}
			}
		}
		# count_owned_ship = { count = 0 # Vfix opt
		# 	limit = {
		# 		has_ship_flag = behemoth_hatchling_ship
		# 	}
		# }
	}

	default = no

	stop_when_armies_dead = no
	stop_when_ground_combat = no
	abduct_pops = yes

	planet_damage = 0.5
	army_damage = 0.5

	kill_pop_chance = {
		base = 0.15
	}
	min_pops_to_kill_pop = 200
	kill_pop_amount = { min = 0 max = 200 }

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 8
		modifier = {
			factor = 0
			exists = from
			from = {
				NAND = {
					pop_amount > 200
					owner = { is_hostile = root.owner }
				}
			}
			NOT = {
				owner = { has_origin = origin_slavers }
			}
		}
		modifier = {
			factor = 0.01
			exists = from
			exists = owner
			owner = { # Vfix otherwise makes no sense
				has_claim = from.solar_system
				NOT = { has_origin = origin_slavers }
			}
		}
		modifier = {
			factor = 100
			owner = { has_origin = origin_slavers }
		}
	}
}

		}
	}

	default = no

	stop_when_armies_dead = no
	stop_when_ground_combat = no
	abduct_pops = no

		base = 1.0
		modifier = {
			factor = 2

				is_country_type = voidworms

		modifier = {

	}
}
