# Prev = planet
CmtEffectCreateFleet_KhanArmadaVoidborn = {
	create_marauder_merc_admiral = yes

	create_fleet = {
		name = "NAME_Armada_of_the_Voidborn"
		effect = {
			set_owner = prev
			create_armada_of_the_voidborn = yes
			inline_script = "marauder/CmtFleetSetAuxParam_KhanCombat"
		}
		settings = {
			can_upgrade = no
			can_disband = no
			can_change_composition = no
			can_change_leader = no
			uses_naval_capacity = no
		}
	}
}
