building_xeno_zoo = {
	base_buildtime = @b1_time
	planet_limit = 1
	category = research

	building_sets = {
		urban
		research
		society
		zoo
	}

	potential = {
		has_owner = yes
		owner = { is_regular_empire = yes is_guided_sapience_empire = no }
		OR = {
			has_modifier = resort_colony
			has_deposit = d_alien_pets_deposit
			has_deposit = d_avian_reserve
			AND = {
				has_owner = yes
				owner = { has_policy_flag = pre_sapients_protect }
				any_owned_pop_group = { is_sapient = no }
			}
			AND = {
				has_owner = yes
				any_owned_pop_group = {
					has_citizenship_type = { type = citizenship_protected_drone country = owner }
				}
			}
			any_owned_pop_group = { is_livestock = yes }
			AND = {
				has_owner = yes
				any_owned_pop_group = { has_trait = trait_nascent_stage }
			}
		}
	}

	destroy_trigger = {
		OR = {
			AND = {
				has_owner = yes
				owner = {
					OR = {
						is_gestalt = yes
						is_guided_sapience_empire = yes
					}
				}
			}
			NOR = {
				has_modifier = resort_colony
				has_deposit = d_alien_pets_deposit
				has_deposit = d_avian_reserve
				AND = {
					has_owner = yes
					owner = { has_policy_flag = pre_sapients_protect }
					any_owned_pop_group = { is_sapient = no }
				}
				AND = {
					has_owner = yes
					any_owned_pop_group = {
						has_citizenship_type = { type = citizenship_protected_drone country = owner }
					}
				}
				any_owned_pop_group = { is_livestock = yes }
				AND = {
					has_owner = yes
					any_owned_pop_group = { has_trait = trait_nascent_stage }
				}
			}
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	inline_script = {
		script = jobs/biologists_add
		AMOUNT = @building_static_jobs
	}

	triggered_planet_modifier = {
		potential = { has_owner = yes has_modifier = resort_colony }
		job_biologist_add = @building_static_jobs
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			OR = {
				any_owned_pop_group = { is_livestock = yes is_lithoid = no }
				any_owned_pop_group = {
					has_citizenship_type = { type = citizenship_protected_drone country = owner }
				}
			}
			NOT = { owner = { has_valid_civic = civic_warrior_culture } }
		}
		job_livestock_zoo_animal_add = 300
	}
	triggered_planet_modifier = {
		potential = {
			any_owned_pop_group = { is_livestock = yes is_lithoid = yes }
			has_owner = yes
			NOT = { owner = { has_valid_civic = civic_warrior_culture } }
		}
		job_livestock_zoo_animal_lithoid_add = 300
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			OR = {
				any_owned_pop_group = { is_livestock = yes is_lithoid = no }
				any_owned_pop_group = {
					has_citizenship_type = { type = citizenship_protected_drone country = owner }
				}
			}
			owner = { has_valid_civic = civic_warrior_culture }
		}
		job_livestock_zoo_beast_add = 300
	}
	triggered_planet_modifier = {
		potential = {
			any_owned_pop_group = { is_livestock = no is_lithoid = yes }
			has_owner = yes
			owner = { has_valid_civic = civic_warrior_culture }
		}
		job_livestock_zoo_beast_lithoid_add = 300
	}

	triggered_desc = {
		trigger = {
			has_owner = yes
			NOT = { owner = { has_valid_civic = civic_warrior_culture } }
		}
		text = pre_sapient_job_xeno_zoo_animal_tt
	}
	triggered_desc = {
		trigger = {
			has_owner = yes
			owner = { has_valid_civic = civic_warrior_culture }
		}
		text = pre_sapient_job_xeno_zoo_beast_tt
	}
	triggered_desc = {
		text = biologist_is_xeno_zoo_keeper_tt
	}
	triggered_desc = {
		text = NEW_LINE
	}
	triggered_desc = {
		trigger = {
			has_owner = yes
			NOT = { owner = { has_valid_civic = civic_warrior_culture } }
		}
		text = job_xeno_zoo_animal_effect_desc
	}
	triggered_desc = {
		trigger = {
			has_owner = yes
			owner = { has_valid_civic = civic_warrior_culture }
		}
		text = job_xeno_zoo_beast_effect_desc
	}

	prerequisites = {

	}
