# Astral Planes Unique Buildings

building_astral_siphon_1 = {
	base_buildtime = @b1_time
	planet_limit = 1
	category = research

	building_sets = {
		research
		physics
	}

	potential = {
		exists = owner
		owner = {
			check_variable = { which = astral_siphon_unlocked value >= 1 }
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		cost = {
			trigger = {
				owner = { is_wilderness_empire = yes }
			}
			biomass = @b2_biomass
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	inline_script = {
		script = buildings/on_all_wilderness_buildings_districts
	}

	country_modifier = {
		country_resource_max_astral_threads_add = 200
	}

	planet_modifier = {
		planet_physicists_physics_research_produces_add = 0.20
		planet_physicists_astral_threads_produces_add = 0.20
	}

	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = { is_gestalt = no }
		}
		planet_physicists_consumer_goods_upkeep_add = 1
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = { is_machine_empire = yes }
		}
		planet_physicists_energy_upkeep_add = 2
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = { is_hive_empire = yes }
		}
		planet_physicists_minerals_upkeep_add = 2
	}

	triggered_desc = {
		text = physicist_is_astral_researcher_tt
	}

	upgrades = {
		building_astral_siphon_2
	}
}
