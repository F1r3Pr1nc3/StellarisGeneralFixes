building_medical_2 = {
	base_buildtime = @b2_time
	can_build = no
	category = amenity

	building_sets = {
		government
		urban
		medical
	}

	potential = {
		has_owner = yes
		# check the empire is valid
		owner = {
			# rule out gestalts and pre-FTL societies
			is_individual_machine = no
			OR = {
				is_regular_empire = yes
				AND = {
					has_origin = origin_fear_of_the_dark
					is_primitive = yes
				}
			}
			# rule out AI empires with entirely no access to "Pharma State" Civic, unless the planet is unstable
			OR = {
				is_ai = no
				has_valid_civic = civic_private_healthcare_corporate
				root = {
					OR = { # Vfix opt
						has_branch_office = no
						branch_office_owner = { has_valid_civic = civic_private_healthcare_corporate }
						NOR = {
							has_available_jobs = entertainer
							has_forbidden_jobs = entertainer
							planet_stability >= 50
							free_amenities > 0
						}
					}
				}
			}
		}
	}

	allow = {
		is_occupied_flag = no
	}

	destroy_trigger = {
		has_owner = yes # Vfix opt
		OR = {
			# if the planet is owned by a gestalt or pre-FTL society
			owner = {
				OR = {
					is_gestalt = yes
					AND = {
						is_primitive = yes
						NOT = { has_origin = origin_fear_of_the_dark }
					}
				}
			}
			# if this is an AI-controlled empire with entirely no access to "Pharma State" Civic, assuming the planet is now stable
			AND = {
				owner = {
					is_ai = yes
					NOT = { has_valid_civic = civic_private_healthcare_corporate }
				}
				NOR = { # Vfix (dissolved unnecessary prev)
					AND = {
						has_branch_office = yes
						branch_office_owner = { has_valid_civic = civic_private_healthcare_corporate }
					}
					planet_stability <= 70
					free_amenities <= 2000 # Vfix?
				}
			}
		}
	}

	inline_script = {
		script = jobs/healthcare_add
		AMOUNT = @building_static_jobs
	}

	planet_modifier = {
		planet_doctors_upkeep_mult = -0.20
		pop_amenities_usage_organic_mult = -0.10
	}

	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = { has_technology = tech_gene_tailoring }
			any_owned_species = { has_auto_modding_trait = yes }
		}
		auto_mod_monthly_add = 100
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = { game_start_auto_mod_empire = yes }
			any_owned_species = { has_auto_modding_trait = yes }
		}
		auto_mod_monthly_add = 100
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = {
				has_country_flag = clone_army_full_potential
				has_active_tradition = tr_cloning_artificial_population_clone_army
			}
		}
		clone_soldiers_sustained = @clones_sustained_low
		mult = 2
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = {
				OR = {
					has_purity_heightened_attributes_tradition = yes
					has_cloning_artificial_population_tradition = yes
					has_mutation_somatic_subversiveness_tradition = yes

			energy = @b2_upkeep
		}
		upkeep = {
			trigger = {
				num_zones = { type = zone_payback_enlightenment value = 0 }
			}
			exotic_gases = @b2_rare_upkeep
		}
	}

	prerequisites = {
		"tech_frontier_health"
	}

	on_destroy = {
		owner = {
			if = {
				limit = { has_event_chain = payback_out_of_warranty_chain }
				end_event_chain = payback_out_of_warranty_chain
			}
		}
	}

	on_repaired = {
		if = {
			limit = {
				owner = { has_event_chain = payback_out_of_warranty_chain }
			}
			random_list = {
				50 = { # still broken
					owner = {
						country_event = { id = origin.4302 }
					}
					modifier = {
						factor = 0
						owner = { has_technology = tech_frontier_health }
					}
				}
				20 = { # repaired
					owner = {
						country_event = { id = origin.4301 }
					}
					modifier = {
						factor = 2
						owner = { num_research_agreements = 1 }
					}
					modifier = {
						factor = 3
						owner = { num_research_agreements = 2 }
					}
					modifier = {
						factor = 4
						owner = { num_research_agreements >= 3 }
					}
				}
				10 = { # broken beyond repair
					planet_event = { id = origin.4303 }
				}
			}
		}
	}

	upgrades = {
		building_medical_3
	}
}
