
triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			# is_regular_empire = yes
			NOR = {
				is_gestalt = yes # Vfix opt
				is_fallen_empire = yes # Do not grant both Enforcers and Protectors from the same source.
				has_valid_civic = civic_dystopian_society
			}
		}
	}

	job_enforcer_add = $AMOUNT$
}

		exists = owner
		owner = { is_gestalt = no is_fallen_empire = yes }
	}

	modifier = {
		job_fe_protector_add = $AMOUNT$
	}
}
