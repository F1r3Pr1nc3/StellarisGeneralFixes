# for use primarily with limited ship types, since this is an expensive check.
hidden_effect = {
	solar_system = {
		every_ship_in_system = {
			limit = {
				controller = {
					is_hostile_to = from.controller
				}
				distance = {
					source = root
					max_distance <= $RANGE$
					same_solar_system = yes
				}
			}
			damage_ship = $DAMAGE$
		}
	}
}
