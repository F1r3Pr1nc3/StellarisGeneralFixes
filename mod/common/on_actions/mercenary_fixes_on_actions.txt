# A planets controller becomes a country not the same as the owner.
# Root = Planet
# From = Planet Owner
# fromfrom = Planet Controller (the one occupying)
on_planet_occupied = {
	events = {
		marauder.30 # Mercenary need new target
	}
}

# This = subject
# From = subject's overlord
on_becoming_subject = {
	events = {
		khan.530 # Stuck Vfix
	}
}

# This = destroyed country
# From = optional, destroyer (country)
on_country_destroyed = {
	events = {
		marauder.600 # Awakened Marauders dead
	}
}

# This = Ship
# From = Disabler Ship
on_ship_disabled = {
	events = {
		pirate.65
	}
}

# This = Ship
on_ship_enabled = {
	events = {
		pirate.66
	}
}

# You have changed the location of your capital. Also called when a country's capital is first set e.g. During galaxy creation (but then FROM is not set)
# this/root = new capital
# from = old capital
on_capital_changed = {
	events = {
		khan.320
	}
}
