	text = {
		trigger = {
			country_uses_food = yes

		}
		localization_key = r_minerals_RULOC_ACC
	}
	text = {
		trigger = {
			country_uses_food = no
		}
		localization_key = r_energy_RULOC_ACC
	}
}

defined_text = {
	name = SubjectResearcherUpkeep_RULOC_ACC
	text = {
		trigger = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		localization_key = r_minerals_RULOC_ACC
	}
	text = {
		trigger = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		localization_key = r_energy_RULOC_ACC
	}
	text = {
		trigger = {
			exists = owner
			owner = {
				is_hive_empire = no
				is_machine_empire = no
			}
		}
		localization_key = r_consumer_goods_RULOC_ACC
	}
}
